Using UI components in Angular

Example: lf-checklist

In the example below, the implementation comes from the CDN and the types are provided in the @laserfiche/types-lf-ui-components NPM package.

  1. Add the following script tag to index.html.

    Insert this script tag after <body>. This script requires Zone.js to load before it loads, and Angular by default loads Zone.js in polyfills.js inside <body>. 

     <script src="https://cdn.jsdelivr.net/npm/@laserfiche/lf-ui-components@NPM_VERSION/cdn/lf-ui-components.js" defer></script>

    If for some reason you do not already import zone.js in polyfills.ts you can add the script tag: 

         <script src="https://cdn.jsdelivr.net/npm/zone.js@0.11.4/bundles/zone.umd.min.js"></script>

  2. Choose a theme to style your components: 

     <link href="https://cdn.jsdelivr.net/npm/@laserfiche/lf-ui-components@NPM_VERSION/cdn/lf-laserfiche-lite.css" rel="stylesheet" />

    or 

     <link href="https://cdn.jsdelivr.net/npm/@laserfiche/lf-ui-components@NPM_VERSION/cdn/lf-ms-ofice-lite.css" rel="stylesheet" />

  3. Add the Angular Material stylesheet 

     <link href="https://cdn.jsdelivr.net/npm/@laserfiche/lf-ui-components@NPM_VERSION/cdn/indigo-pink.css" rel="stylesheet" />
  4. Install the types NPM package. 
     npm install @laserfiche/types-lf-ui-components@NPM_VERSION

  5. Inside ./app.module.ts, add CUSTOM_ELEMENTS_SCHEMA to @NgModule.schemas. This allows you to use custom non-Angular components in your Angular application. 

    import {  NgModule, CUSTOM_ELEMENTS_SCHEMA } from '@angular/core';
@NgModule({
  declarations: [
    AppComponent,
    ...
  ],
  imports: [
    ...
  ],
  schemas: [CUSTOM_ELEMENTS_SCHEMA]
})

  6. Add to a component

    In the html file,

    <lf-checklist #myChecklist (checklistChanged)="onChecklistChanged($event)"></lf-checklist>
<!-- checklistChanged is a HTML custom event -->

    In the .ts file,

    import { AfterViewInit, Component, ViewChild, ElementRef } from '@angular/core';
// we can directly import types from the NPM package
import {LfChecklistComponent, Checklist } from '@laserfiche/types-lf-ui-components';

@Component({
  selector: 'app-attachments',
  templateUrl:  './attachments.component.html',
  styleUrls: ['./attachments.component.css'],
})

export class ExampleApp implements AfterViewInit {

  // allows access to the LfChecklistElement in the html above, available starting in ngAfterViewInit hook
  @ViewChild('myChecklist') checklist!: ElementRef<LfChecklistComponent>;

  // type Checklist is the type of the custom element imported from the NPM package
  currentChecklist: Checklist[];

  // events for the elements are received as CustomEvents 
  onChecklistChanged(event: CustomEvent<Checklist>) {
    const checklist = event.detail;    // data will be stored in the detail property
  }

  ngAfterViewInit() {
    // the element's view is initialized after ngAfterViewInit
    this.checklist.nativeElement.initAsync(...);
  }
}
