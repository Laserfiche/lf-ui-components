Example: lf-checklist In an Angular Project Via CDN

In the example below, the implementation comes from the CDN and the types is provided in the @laserfiche/types-lf-ui-components NPM package.

  1. Add the following script tags to index.html 

     <script  src="https://unpkg.com/@laserfiche/lf-ui-components@NPM_VERSION/cdn/laserfiche-ui-components.js"defer></script>
 <link href="https://unpkg.com/@laserfiche/lf-ui-components@NPM_VERSION/cdn/lf-laserfiche-lite.css"rel="stylesheet" />
 <link href="https://unpkg.com/@laserfiche/lf-ui-components@NPM_VERSION/cdn/indigo-pink.css"rel="stylesheet" />
  2. Install the types definition for the custom elements. 
     npm install @laserfiche/types-lf-ui-components@NPM_VERSION

  3. Inside ./app.module.ts, add CUSTOM_ELEMENTS_SCHEMA to @NgModule.schemas. This enables custom elements in your Angular application. 

    import {  NgModule, CUSTOM_ELEMENTS_SCHEMA } from '@angular/core';
@NgModule({
declarations: [
  AppComponent,
  ...
],
imports: [
  ...
],
schemas: [CUSTOM_ELEMENTS_SCHEMA]
})

  4. In a html file,

    <lf-checklist #myChecklist (checklistChanged)="onChecklistChanged($event)"></lf-checklist>
<!-- checklistChanged is a HTML custom event -->

    To add a handle for this element, in the .ts file,

    import { AfterViewInit, Component, ViewChild, ElementRef } from '@angular/core';
// we can directly import types from the NPM package
import {LfChecklistComponent, Checklist } from '@laserfiche/types-lf-ui-components';

@Component({
selector: 'app-attachments',
templateUrl:  './attachments.component.html',
styleUrls: ['./attachments.component.css'],
})

export class ExampleApp implements AfterViewInit {

// allows access to the LfChecklistElement in the html above, available starting in ngAfterViewInit hook
@ViewChild('myChecklist') checklist!: ElementRef

// type Checklist is the type of the custom element imported from the NPM package
currentChecklist: Checklist[];

// events for the elements are received as CustomEvents 
onChecklistChanged(event: CustomEvent <Checklist>	) {
  const checklist = event.detail;    // data will be stored in the detail property
}

ngAfterViewInit() {
// the element's view is initialized after ngAfterViewInit
this.checklist.nativeElement.initAsync(...)
}
}

Example: lf-checklist In a ReactJS Project Via CDN

In the example below, the implementation comes from the CDN.

  1. Add the following script tags to ./public/index.html 

     <script  src="https://unpkg.com/@laserfiche/lf-ui-components@NPM_VERSION/cdn/laserfiche-ui-components.js"defer></script>
 <link href="https://unpkg.com/@laserfiche/lf-ui-components@NPM_VERSION/cdn/lf-laserfiche-lite.css"rel="stylesheet" />
 <link href="https://unpkg.com/@laserfiche/lf-ui-components@NPM_VERSION/cdn/indigo-pink.css"rel="stylesheet" />

  2. In the .js file,

    import React from 'react';
export class App extends React.Component {
  checklist = React.createRef();  // create handle for custom element

  constructor(props) {
    super(props);
  }

  async componentDidMount() {
    await this.checklist.current?.initAsync(...);
    this.checklist.current?.addEventListener('checklistChanged', this.onChecklistChanged);
    // we have to add event listen to listen for the HTML custom event checklistChanged
  }

  onChecklistChanged(event) {
    const checklist = event.detail;    // data will be stored in the detail property
  }

  render() {
  return (
    <lf-checklist
      ref={this.checklist}
      ></lf-checklist>
    );
  }
}
